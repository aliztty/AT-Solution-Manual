\section{گروه بنیادی}
\linespread{1.5}\selectfont

\begin{problem}{1.5.1}
    فرض کنید $X$ یک فضای توپولوژیک با توپولوژی گسسته است.
    درباره $\pi_1\left(X,x_0\right)$ که در آن $x_0 \in X$ یک نقطه دلخواه است چه می توان گفت؟
    راهنمایی. از لم ۲.۱.۱ به همراه تمرین ۲.۴.۱ استفاده کنید.
    
    \solution
    \textbf{لم}: فرض کنید $C$ فضای توپولوژی دلخواه باشد. آنگاه هر تابع پیوسته $f:C \rightarrow X$ تابع ثابت خواهد بود.
    
    \textbf{برهان لم}: به برهان خلف فرض کنید این چنین نباشد، یعنی تصویر تابع دارای بیشتر از یک عضو خواهد بود و قرار دهید $x_1 \in \text{Im}(f)$ آنگاه طبق فرض خلف $\text{Im}(f) - \{x_1\} \neq \emptyset$. حال $U = \{x_1\}$ و $V = \text{Im}(f) - \{x_1\}$ یک جداسازی برای $\text{Im}(f)$ خواهد بود که با فرض تناقض دارد.
    
    حال قرار دهید $C = I$ و $g, f \in \pi_1\left(X, x_0\right)$ دلخواه باشند. آنگاه داریم:
    \[ f(0) = g(0) = x_0 \quad \Longrightarrow \quad f = C_{x_0} = g_{x_0} \]
    در نتیجه $\pi_1\left(X,x_0\right) \cong \{e\}$ یکریخت با گروه بدیهی خواهد بود.
\end{problem}

\begin{problem}{1.5.2}
    فرض کنید $G$ یک گروه توپولوژیک مسیری همبند باشد و $e \in G$ عضو خنثی این گروه باشد. نشان دهید $\pi_1\left(G,e\right)$ یک گروه آبلی است. برای تعریف و مثال هایی در مورد گروه های توپولوژیک فصل ۵ را ببینید.
    
    \solution
    تعریف می کنیم $\circ : \pi_1(G, e) \times \pi_1(G, e) \rightarrow \pi_1(G, e)$ آنگاه داریم:
    \[ \left(f \circ g\right)(t) := f(t) \cdot g(t) \]
    اگر $f,g$ اعضای گروه بنیادی $\pi_1\left(G,e\right)$ باشد آنگاه $f \circ g: I\rightarrow X$ پیوسته خواهد بود چون عملگر گروه پیوسته است.
    
    \begin{equation*}
        \left.
        \begin{array}{ll}
            (f \circ g)(0) = f(0) \cdot g(0) = e \cdot e = e    \\
            (f \circ g)(1) = f(1) \cdot g(1) = e \cdot e = e
        \end{array}
        \right\} \Rightarrow f\circ g \in \pi_1\left(G,e\right)
    \end{equation*}
    
    اگر $f_1, f_2, g_1, g_2 \in \pi_1\left(G,e\right)$:
    \[
        \left(f_1 \ast g_1\right) \circ \left(f_2 \ast g_2 \right) = 
        \left\{
        \begin{array}{ll}
            f_1 \cdot f_2 & 0 \le t \le \frac{1}{2}    \\
            g_1 \cdot g_2 & 0 \le t \le 1    
        \end{array}
        \right. = \left( f_1 \circ f_2 \right) \ast \left( g_1 \circ g_2 \right)
    \]
    
    از طرف دیگر، داریم $C_{e} \ast f \cong f \ast C_{e} \cong f$ که نتیجه می دهد:
    \begin{align*}
        \left( f \ast C_e \right) \circ \left( C_e \ast g \right) &= \left( f \circ C_e \right) \ast \left( C_e \circ g\right) = f \ast g \\
        \left( C_e \ast f \right) \circ \left( g \ast C_e \right) &= \left( C_e \circ g \right) \ast \left( f \ast C_e \right) = g \ast f
    \end{align*}
    
    توجه داریم که برای هر انتخاب $f,g \in \pi_1\left(G,e\right)$ اگر گزاره زیر را اثبات کنیم کار تمام است:    
    \[
        \left.
        \begin{array}{c}
            f \cong f' \\
            g \cong g'
        \end{array}
        \right\} \Rightarrow f\circ g \cong f' \circ g'
    \]
    
    فرض کنید $H , G$ به ترتیب هموتوپی های متناظر برای $(f, f')$ و $(g, g')$ باشند. ادعا می کنیم که $H \circ G$ هموتوپی بین $f \circ g$ و $f'\circ g'$ است.
    
    \textbf{بررسی پیوستگی}: پیوستگی $H$ و $G$ پیوستگی $H\circ G$ را تضمین می کند.
    
    \textbf{هموتوپی بودن}:
    \begin{align*}
        \left( H \circ G \right)(s, 0) &= H(s,0) \cdot G(s,0) = f \circ g \\
        \left( H \circ G \right)(s, 1) &= H(s,1) \cdot G(s,1) = f' \circ g'
    \end{align*}
    
    در نتیجه $f \circ g \cong f' \circ g'$، و در نتیجه $\circ$ خوش تعریف است. آنگاه:
    \[
    \left.
    \begin{array}{c}
        f \ast C_e \cong f \cong C_e \ast f \\
        C_e \ast g \cong g \cong g \ast C_e
    \end{array}
    \right\} \Rightarrow f \ast g \cong g \ast f 
    \]
    \[
    \Rightarrow [f] \cdot [g] = [f \ast g] = [g \ast f] = [g] \cdot [f]
    \]
    این اثبات را کامل می کند.
\end{problem}

\begin{problem}{1.5.3}
    \begin{itemize}
        \item[(الف)] آیا می توان گفت که گروه بنیادی همواره یک گروه شمارش پذیر است؟ در مورد گروه بنیادی گوشواره نامتناهی اشاره شده در مثال ۳.۶.۶ چه می توان گفت؟
        \item[(ب)] با فرض
        \[ \pi_1\left(X,x_0\right) \simeq \pi_0\left( \Omega\left(X,x_0\right),\omega_0\right) \]
        که در آن $\omega_0:I\rightarrow X$ مسیر ثابت بر پایه $x_0$ است، آیا می توانید یک مثال از فضایی با گروه ناشمارا بیابید؟
        \item[(پ)] آیا می توان گفت برخی شرایط، شمارش پذیر بودن گروه بنیادی یک فضا را تضمین می کند؟
    \end{itemize}
    
    \solution
    \begin{itemize}
        \item[(الف)] \textbf{خیر}.
        فرض کنید $n \in \mathbb{N}$ آنگاه $C_n$ را تعریف می کنیم:
        \[ C_n = \left\{ (x,y) \in\mathbb{R}^{2} : \left(x-\frac{1}{n}\right)^{2} + y^2 = \left(\frac{1}{n}\right)^{2}\right\} \]
        در این صورت مجموعه $\mathcal{H}$ گوشواره نامتناهی نامیده می شود.
        \[ \mathcal{H} := \bigcup_{n=1}^{\infty} C_n \]
        حال به محاسبه گروه بنیادی این فضا با در نظر گرفتن مبدا مختصات به عنوان نقطه پایه می پردازیم ($\pi_1\left(\mathcal{H},0\right)$).
        برای هر دنباله به فرم $\left(a_n\right)_{n\in\mathbb{N}} \in \{0,1\}^{\mathbb{N}}$، لوپی تعریف می کنیم به فرم $\alpha_{\left(a_n\right)} : \left[0,1\right] \rightarrow \mathcal{H}$ به طوری که:
        \[
            \left\{
            \begin{array}{ll}
            \text{به دور } C_n \text{ بچرخد} & a_n = 1 \\
            \text{در نقطه پایه بماند} & a_n = 0
            \end{array}
            \right.
        \]
        یک تناظر یک به یک بین دنباله های تشکیل شده از ۰ و ۱ و این لوپ ها موجود است که در این صورت واضحا تعداد این لوپ ها ناشمارا خواهد بود.
        
        \item[(ب)] با توجه به بخش قبلی حداقل یک مثال ارائه شده داریم.
        در نظر داشته باشید که تعریف توپولوژی فشرده-باز با پایه هایش به فرم زیر است:
        $(K \underset{\text{\tiny فشرده}}{\subseteq} I,\ U \underset{\text{\tiny باز}}{\subseteq} X) $ 
        \[ S\left(K,U\right)  = \left\{ f:I\rightarrow X : f\left(K\right) \subseteq U \right\} \]
        در نتیجه مولفه های همبندی، مسیر های بسته ای هستند که هموتوپ می باشند. در نتیجه $\pi_0$ تعداد مولفه های همبندی را می شمارد.
        
        \item[(پ)] منیفلد بودن شرط کافی است.
        برهان. ویژگی ۱.۱۶ \cite{lee_smooth_manifolds}

\section{محاسبه گروه بنیادی}

\begin{problem}{2.3.1}
    \begin{itemize}
        \item[(الف)] فرض کنید $i \in \{1,2\}$ و $p_i:E_i\rightarrow B_i$ دو نگاشت پوششی باشند. نشان دهید:
        \[ p_1 \times p_2 : E_1 \times E_2 \rightarrow B_1 \times B_2 \]
        یک نگاشت پوششی است.
        \item[(ب)] آیا گزاره بخش (الف) به حاصل ضرب های دلخواه قابل تعمیم است وقتی فضاهای حاصل ضربی مجهز به توپولوژی حاصل ضربی باشند؟
    \end{itemize}

    \solution
    \begin{itemize}
        \item[(الف)] فرض کنید $b = \left(b_1, b_2\right) \in B_1 \times B_2$. از آنجایی که $p_1$ و $p_2$ نگاشت پوششی هستند، همسایگی های بازی مانند $U_1 \subset B_1$ شامل $b_1$ و $U_2 \subset B_2$ شامل $b_2$ وجود دارند. واضحا تجزیه به مجموعه های باز مجزا داریم:
        \begin{align*}
            p_1^{-1}\left(U_1\right) &= \bigsqcup_{\alpha \in \Lambda} V_{\alpha} \\
            p_2^{-1}\left(U_2\right) &= \bigsqcup_{\beta \in \Gamma} W_{\beta}
        \end{align*}
        که در آن تحدید های $p_1 |_{V_{\alpha}}:V_{\alpha}\rightarrow U_1 $ و $p_2 |_{W_{\beta}}:W_{\beta}\rightarrow U_2 $ همریختی هستند.
        
        مجموعه باز $U = U_1 \times U_2$ را در توپولوژی حاصل ضربی در نظر بگیرید. تصویر وارون آن عبارت است از:
        \[ \left(p_1 \times p_2\right)^{-1}\left(U_1 \times U_2\right) = \left( \bigsqcup_{\alpha\in\Lambda} V_{\alpha} \right) \times \left( \bigsqcup_{\beta\in\Gamma} W_{\beta} \right) \]
        با استفاده از خاصیت توزیع پذیری حاصل ضرب دکارتی بر روی اجتماع های مجزا:
        \[ \bigsqcup_{(\alpha,\beta)\in\Lambda \times \Gamma} \left(V_{\alpha} \times W_{\beta} \right) \]
        برای هر زوج $(\alpha, \beta)$، نگاشت تحدید حاصل ضربی از همریختی ها است که خود یک همریختی است. بنابراین $U_1 \times U_2$ یک همسایگی به طور هموار پوشیده شده است.
        
        \item[(ب)] \textbf{خیر}. انتظار داریم در حالت نامتناهی به مشکل بر بخوریم.
        مثال نقض: فرض کنید $p : \mathbb{R} \rightarrow S^{1}$ نگاشت پوششی استاندارد تعریف شده با $p(t) = \left( \cos{(2\pi t)} , \sin{(2\pi t)} \right)$ باشد. حاصل ضرب شمارای این نگاشت با خودش را در نظر بگیرید:
        \[ P : \prod_{n\in\mathbb{N}} \mathbb{R} \rightarrow \prod_{n\in \mathbb{N}} S^1 \]
        فرض کنید $B = \prod S^1$ باشد. در توپولوژی حاصل ضربی، هر مجموعه باز پایه $U \subset B$ به فرم $U = \prod U_n$ است، که در آن برای همه اندیس های $n$ به جز تعدادی متناهی، $U_n = S^1$ می باشد.
        برای اینکه $U$ به طور هموار پوشیده باشد، $P^{-1}(U)$ باید اجتماع مجزایی از مجموعه های باز باشد. با این حال، با توجه به هر عامل $k$ که در آن $U_k = S^1$ است، تصویر وارون برابر است با $\mathbb{R}$. چون $\mathbb{R}$ همبند است، اما نگاشت یک به یک نیست، $P$ نمی تواند هیچ برشی را به طور همریخت نگاشت کند. در نتیجه $P$ نگاشت پوششی نیست.

    \end{itemize}
\end{problem}
